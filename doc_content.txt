6.1 用户与权限管理
模块
（脚本完成人：王挚）
索引设计思路：
用户日常使用中最频繁的操作是查看“我的申请”或辅导员查看“待我审批”的记录。经常需要根据申请人学工号、审批人学工号以及审批状态进行条件筛选。因此，在审批流程总表上针对这三个字段建立普通索引，以加速流程检索。
索引创建 SQL 脚本：
-- 审批流程总表索引
CREATE INDEX idx_approval_applicant ON 审批流程总表(申请人学工号);
CREATE INDEX idx_approval_approver ON 审批流程总表(审批人学工号);
CREATE INDEX idx_approval_status ON 审批流程总表(审批状态);

索引设计思路：

CREATE INDEX idx_task_manager ON 日常任务表(负责人学工号);
6.3 班费收支管理模块（脚本完成人：陈柔伊）  
索引设计思路：
财务报表按月查询：班委在进行班费对账时，高频操作是按时间段（如某个月）汇总收入和支出。因此，需要在班费收入记录表的收入日期字段，以及班费支出申请表的申请日期字段建立索引，优化范围查询性能。
学生个人缴费核对：需要快速定位某个学生的缴费情况，在学生缴费明细表的学工号上创建索引。
索引创建 SQL 脚本：
-- 班费收支日期索引
CREATE INDEX idx_income_date ON 班费收入记录表(收入日期);
CREATE INDEX idx_expense_date ON 班费支出申请表(申请日期);
-- 学生缴费明细索引
CREATE INDEX idx_payment_student ON 学生缴费明细表(学工号);

 学业管理模块（脚本完成人：杨棋冰） 
索引设计思路：
成绩单查询：学生期末查询成绩或辅导员拉取某学期全班成绩单是高频并发操作。查询条件高度依赖学生学工号和学期。为了避免全表扫描，在学业成绩明细表上建立联合索引（学生学工号，学期）。
索引创建 SQL 脚本：
-- 学业成绩联合索引
CREATE INDEX idx_score_student_semester ON 学业成绩明细表(学生学工号, 学期);

7.1.1 
用户与权限管理模块
（
脚本完成人：王挚
）
以下为四个全局公共表：
-- 1. 角色表
CREATE TABLE 角色表 (
  角色编码 VARCHAR(50) PRIMARY KEY COMMENT '角色编码',
  角色名称 VARCHAR(100) NOT NULL COMMENT '角色名称',
  权限配置清单 TEXT COMMENT '权限配置清单',
  创建时间 DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  最后调整时间 DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '最后调整时间'
) ENGINE=InnoDB;
-- 2. 用户账号与权限信息表
CREATE TABLE 用户账号与权限信息表 (
  学工号 VARCHAR(50) PRIMARY KEY COMMENT '学工号',
  姓名 VARCHAR(50) NOT NULL COMMENT '姓名',
  所属班级 VARCHAR(100) COMMENT '所属班级',
  角色编码 VARCHAR(50) NOT NULL COMMENT '角色编码',
  角色名称 VARCHAR(100) COMMENT '角色名称',
  联系方式 VARCHAR(50) COMMENT '联系方式',
  登录密码 VARCHAR(255) NOT NULL COMMENT '登录密码',
  账号状态 VARCHAR(20) DEFAULT '正常' COMMENT '账号状态',
  权限配置明细 TEXT COMMENT '权限配置明细',
  创建时间 DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  最后登录时间 DATETIME COMMENT '最后登录时间',
  最后更新时间 DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '最后更新时间',
  FOREIGN KEY (角色编码) REFERENCES 角色表(角色编码) ON DELETE RESTRICT ON UPDATE CASCADE
) ENGINE=InnoDB;
-- 3. 审批流程总表
CREATE TABLE 审批流程总表 (
  审批编号 VARCHAR(50) PRIMARY KEY COMMENT '审批编号',
  审批类型 VARCHAR(50) NOT NULL COMMENT '审批类型',
  申请人学工号 VARCHAR(50) NOT NULL COMMENT '申请人学工号',
  申请人姓名 VARCHAR(50) COMMENT '申请人姓名',
  申请内容 TEXT NOT NULL COMMENT '申请内容',
  附件材料 VARCHAR(255) COMMENT '附件材料',
  申请时间 DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '申请时间',
  当前审批节点 VARCHAR(100) COMMENT '当前审批节点',
  审批人学工号 VARCHAR(50) COMMENT '审批人学工号',
  审批人姓名 VARCHAR(50) COMMENT '审批人姓名',
  审核意见 TEXT COMMENT '审核意见',
  审批状态 VARCHAR(20) DEFAULT '待审批' COMMENT '审批状态',
  审核时间 DATETIME COMMENT '审核时间',
  流程完结时间 DATETIME COMMENT '流程完结时间',
  关联业务编号 VARCHAR(50) COMMENT '关联业务编号',
  FOREIGN KEY (申请人学工号) REFERENCES 用户账号与权限信息表(学工号) ON DELETE CASCADE,
  FOREIGN KEY (审批人学工号) REFERENCES 用户账号与权限信息表(学工号) ON DELETE SET NULL
) ENGINE=InnoDB;
-- 4. 系统全局操作日志表
CREATE TABLE 系统全局操作日志表 (
  操作编号 VARCHAR(50) PRIMARY KEY COMMENT '操作编号',
  操作人学工号 VARCHAR(50) NOT NULL COMMENT '操作人学工号',
  操作模块 VARCHAR(50) NOT NULL COMMENT '操作模块',
  操作类型 VARCHAR(50) NOT NULL COMMENT '操作类型',
  操作内容 TEXT COMMENT '操作内容',
  操作IP VARCHAR(50) COMMENT '操作 IP',
  操作时间 DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '操作时间',
  操作结果 VARCHAR(20) NOT NULL COMMENT '操作结果',
  关联业务编号 VARCHAR(50) COMMENT '关联业务编号',
  审核状态 VARCHAR(20) COMMENT '审核状态',
  审核意见 TEXT COMMENT '审核意见',
  FOREIGN KEY (操作人学工号) REFERENCES 用户账号与权限信息表(学工号) ON DELETE CASCADE
) ENGINE=InnoDB;
此表为模块专用表：
-- 5. 权限变更申请表
CREATE TABLE 权限变更申请表 (
  申请编号 VARCHAR(50) PRIMARY KEY COMMENT '申请编号',
  申请人学工号 VARCHAR(50) NOT NULL COMMENT '申请人学工号',
  目标用户学工号 VARCHAR(50) NOT NULL COMMENT '目标用户学工号',
  变更内容 TEXT NOT NULL COMMENT '变更内容',
  变更事由 TEXT COMMENT '变更事由',
  申请时间 DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '申请时间',
  审核人学工号 VARCHAR(50) COMMENT '审核人学工号',
  审核状态 VARCHAR(20) DEFAULT '待审核' COMMENT '审核状态',
  审核意见 TEXT COMMENT '审核意见',
  审核时间 DATETIME COMMENT '审核时间',
  关联审批编号 VARCHAR(50) COMMENT '关联审批编号',
  FOREIGN KEY (申请人学工号) REFERENCES 用户账号与权限信息表(学工号) ON DELETE CASCADE,
  FOREIGN KEY (目标用户学工号) REFERENCES 用户账号与权限信息表(学工号) ON DELETE CASCADE,
  FOREIGN KEY (审核人学工号) REFERENCES 用户账号与权限信息表(学工号) ON DELETE SET NULL,
  FOREIGN KEY (关联审批编号) REFERENCES 审批流程总表(审批编号) ON DELETE SET NULL
) ENGINE=InnoDB;

CREATE PROCEDURE sp_新增班费收入并生成缴费明细(
    IN p_收入日期 DATE,
    IN p_收入金额 DECIMAL(10,2),
    IN p_收入类型 VARCHAR(50),
    IN p_缴费方式 VARCHAR(50),
    IN p_备注 TEXT,
    IN p_经手人学工号 VARCHAR(50),
    OUT p_生成的收入编号 VARCHAR(50)
)
BEGIN
DECLARE v_收入编号 VARCHAR(50);
    DECLARE v_学工号 VARCHAR(50);
    DECLARE v_done INT DEFAULT 0;
    DECLARE student_cursor CURSOR FOR
SELECT 学工号 FROM 用户账号与权限信息表 WHERE 角色编码 != 'ROLE_TEACHER';
DECLARE CONTINUE HANDLER FOR NOT FOUND SET v_done = 1;
    
    -- 收入编号格式：INCOME_YYYYMMDD_序号
    SET v_收入编号 = CONCAT('INCOME_', DATE_FORMAT(p_收入日期, '%Y%m%d'), '_', 
        LPAD((SELECT COUNT(*) + 1 FROM 班费收入记录表 WHERE 收入日期 = p_收入日期), 3, '0'));
    
    START TRANSACTION;
    INSERT INTO 班费收入记录表(收入编号, 收入日期, 收入金额, 收入类型, 缴费方式, 备注, 经手人学工号)
    VALUES(v_收入编号, p_收入日期, 0, p_收入类型, p_缴费方式, p_备注, p_经手人学工号);
    
    OPEN student_cursor;
    
    FETCH student_cursor INTO v_学工号;
    
    student_loop: LOOP
        IF v_done = 1 THEN
            LEAVE student_loop;
        END IF;
        
        INSERT INTO 学生缴费明细表(缴费明细ID, 学工号, 收入编号, 缴费金额)
        VALUES(
            CONCAT('PAY_', v_收入编号, '_', v_学工号),
            v_学工号,
            v_收入编号,
            p_收入金额
        );
        FETCH student_cursor INTO v_学工号;
    END LOOP;
    
    CLOSE student_cursor;
    COMMIT;
    SET p_生成的收入编号 = v_收入编号;
    SELECT CONCAT('成功创建班费收入记录，收入编号：', v_收入编号, '，已为', (SELECT COUNT(*) FROM 用户账号与权限信息表 WHERE 角色编码 = 'ROLE_STUDENT'), '名学生生成缴费明细') AS 执行结果;
END //
DELIMITER ;

CREATE PROCEDURE sp_审批班费支出申请(
    IN p_支出编号 VARCHAR(50),
    IN p_审批人学工号 VARCHAR(50),
    IN p_审批结果 VARCHAR(20),
    IN p_审核意见 TEXT
)
BEGIN
    DECLARE v_当前状态 VARCHAR(20);
    DECLARE v_初审人学工号 VARCHAR(50);
    DECLARE v_终审人学工号 VARCHAR(50);
    DECLARE v_审批人姓名 VARCHAR(50);
    DECLARE v_申请人姓名 VARCHAR(50);
    DECLARE v_审批编号 VARCHAR(50);
    DECLARE v_申请人学工号 VARCHAR(50);
    DECLARE v_申请日期 DATE;
    DECLARE v_支出事由 TEXT;
    DECLARE v_票据凭证 VARCHAR(255);
-- 信息获取
    SELECT 姓名 
    INTO v_审批人姓名 
    FROM 用户账号与权限信息表 
    WHERE 学工号 = p_审批人学工号;
    SELECT 审批状态,初审人学工号,终审人学工号,申请人学工号,申请日期,支出事由,票据凭证
    INTO v_当前状态,v_初审人学工号,v_终审人学工号,v_申请人学工号,v_申请日期,v_支出事由,v_票据凭证
    FROM 班费支出申请表 
    WHERE 支出编号 = p_支出编号;
-- 确认状态合法性
    IF v_当前状态 IS NULL THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = '错误：该支出申请不存在';
    END IF;
    IF v_当前状态 NOT IN ('待初审','待终审') THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = '错误：该申请已完成审批或已被拒绝，无法再次审批';
    END IF;
    SELECT 姓名 INTO v_申请人姓名 FROM 用户账号与权限信息表 WHERE 学工号 = v_申请人学工号;
    SET v_审批编号 = CONCAT('AP_', DATE_FORMAT(NOW(), '%Y%m%d'), '_', 
        LPAD((SELECT COUNT(*) + 1 FROM 审批流程总表 WHERE DATE(审核时间) = CURDATE()), 3, '0'));
    START TRANSACTION;
    CASE p_审批结果
        WHEN '初审通过' THEN
            IF v_当前状态 != '待初审' THEN
                SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = '错误：该申请当前不是待初审状态';
            END IF;
            UPDATE 班费支出申请表 
            SET 审批状态 = '待终审',
初审人学工号 = p_审批人学工号 
            WHERE 支出编号 = p_支出编号;
            INSERT INTO 审批流程总表
            (审批编号,审批类型,申请人学工号,申请人姓名,申请内容,附件材料,
            申请时间,当前审批节点,审批人学工号,审批人姓名,审核意见,审批状态,
            审核时间,流程完结时间,关联业务编号)
            VALUES
            (v_审批编号,'班费支出初审',v_申请人学工号,v_申请人姓名,v_支出事由,v_票据凭证,
            v_申请日期,'待终审',p_审批人学工号,v_审批人姓名,p_审核意见,'初审通过',
            NOW(),NULL,p_支出编号);
            SELECT '初审通过，申请已进入终审阶段' AS 执行结果;
        WHEN '终审通过' THEN
            IF v_当前状态 != '待终审' THEN
                SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = '错误：该申请当前不是待终审状态';
            END IF;
            UPDATE 班费支出申请表 
            SET 审批状态 = '已通过',
终审人学工号 = p_审批人学工号,
                支付日期 = CURDATE() 
            WHERE 支出编号 = p_支出编号;
            INSERT INTO 审批流程总表
            (审批编号,审批类型,申请人学工号,申请人姓名,申请内容,附件材料,
            申请时间,当前审批节点,审批人学工号,审批人姓名,审核意见,审批状态,
            审核时间,流程完结时间,关联业务编号)
            VALUES
            (v_审批编号,'班费支出终审',v_申请人学工号,v_申请人姓名,v_支出事由,v_票据凭证,
            v_申请日期,'已通过',p_审批人学工号,v_审批人姓名,p_审核意见,'终审通过',
            NOW(),NOW(),p_支出编号);
            SELECT '终审通过，申请已完成审批，款项将在今日支付' AS 执行结果;
        WHEN '拒绝' THEN

    DECLARE v_normal INT DEFAULT 0;

    END IF;

注：由于

-- 功能：输入用户的学工号，系统通过多条 SQL 语句分布统计该用户分配到的总任务数和已通过验收的任务数，并通过 IF...ELSE 逻辑控制计算出任务完成率，最终返回一行类似表值的统计结果集。
DELIMITER //
CREATE PROCEDURE sp_get_user_task_summary(IN p_user_id VARCHAR(50))
BEGIN
    -- 语句 1：声明多个内部变量（多语句特性体现）
    DECLARE v_total_tasks INT DEFAULT 0;
    DECLARE v_completed_tasks INT DEFAULT 0;
    DECLARE v_completion_rate VARCHAR(20);
    -- 语句 2 & 3：分布查询赋值
    SELECT COUNT(*) INTO v_total_tasks 
    FROM 日常任务表 
    WHERE 负责人学工号 = p_user_id;
    
    SELECT COUNT(*) INTO v_completed_tasks 
    FROM 日常任务表 
    WHERE 负责人学工号 = p_user_id 
      AND 验收状态 = '验收通过';
    -- 语句 4：条件控制与逻辑计算
    IF v_total_tasks > 0 THEN
        -- 计算完成率并拼接百分号
        SET v_completion_rate = CONCAT(ROUND((v_completed_tasks / v_total_tasks) * 100, 2), '%');
    ELSE
        SET v_completion_rate = '0.00%';
    END IF;
    -- 语句 5：返回类似表值函数的结果集
    SELECT 
        p_user_id AS 学工号,
        v_total_tasks AS 分配任务总数,
        v_completed_tasks AS 验收通过数量,
        v_completion_rate AS 任务综合完成率;
END //

函数 1：计算学生学期出勤率（百分比）
DELIMITER //
CREATE FUNCTION fn_student_attendance_rate(
    p_student_id VARCHAR(50),
    p_start_date DATE,
    p_end_date DATE
)
RETURNS DECIMAL(5,2)
DETERMINISTIC
BEGIN
    DECLARE v_total INT DEFAULT 0;
    DECLARE v_normal INT DEFAULT 0;
    DECLARE v_rate DECIMAL(5,2) DEFAULT 0.00;
    
    SELECT 
        SUM(CASE WHEN 考勤状态 = '正常出勤' THEN 1 ELSE 0 END),
        COUNT(*)
    INTO v_normal, v_total
    FROM 纪律与考勤档案表
    WHERE 学生学工号 = p_student_id
      AND 记录类型 = '考勤'
      AND 记录时间 BETWEEN p_start_date AND p_end_date;
    
    IF v_total > 0 THEN
        SET v_rate = (v_normal * 100.0) / v_total;
    END IF;
    
    RETURN ROUND(v_rate, 2);
END //
DELIMITER ;

DELIMITER //
CREATE FUNCTION fn_class_abnormal_students(
    p_class_name VARCHAR(100),
    p_threshold INT
)
RETURNS TABLE (
    学生学工号 VARCHAR(50),
    学生姓名 VARCHAR(50),
    异常次数 INT
)
DETERMINISTIC
BEGIN
    RETURN QUERY
    SELECT 
        u.学工号,
        u.姓名,
        COUNT(*) AS 异常次数
    FROM 纪律与考勤档案表 d
    JOIN 用户账号与权限信息表 u ON d.学生学工号 = u.学工号
